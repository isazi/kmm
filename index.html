KMM

KMM is a lightweight C++ middleware for accelerated computing.

Basic Example

This example shows how to run a CUDA kernel implementing a vector add operation with KMM.

 1#include <deque>
 2#include <iostream>
 3
 4#include "spdlog/spdlog.h"
 5
 6#include "kmm/array.hpp"
 7#include "kmm/cuda/cuda.hpp"
 8#include "kmm/host/host.hpp"
 9#include "kmm/runtime_handle.hpp"
10
11#define SIZE 65536000
12
13__global__ void vector_add(const float* A, const float* B, float* C, int size) {
14    int item = (blockDim.x * blockIdx.x) + threadIdx.x;
15
16    if (item < size) {
17        C[item] = A[item] + B[item];
18    }
19}
20
21void initialize(float* A, float* B) {
22#pragma omp parallel for
23    for (unsigned int item = 0; item < SIZE; item++) {
24        A[item] = 1.0;
25        B[item] = 2.0;
26    }
27
28    std::cout << "initialize\n";
29}
30
31void execute(kmm::CudaDevice& device, float* C, const float* A, const float* B) {
32    int block_size = 256;
33    int num_blocks = (SIZE + block_size - 1) / block_size;
34    vector_add<<<num_blocks, block_size, 0, device.stream()>>>(A, B, C, SIZE);
35}
36
37void verify(const float* C) {
38#pragma omp parallel for
39    for (unsigned int item = 0; item < SIZE; item++) {
40        if (fabsf(C[item] - 3.0f) > 1.0e-9) {
41            std::cout << "ERROR" << std::endl;
42            break;
43        }
44    }
45
46    std::cout << "SUCCESS" << std::endl;
47}
48
49int main(void) {
50    spdlog::set_level(spdlog::level::debug);
51
52    unsigned int threads_per_block = 256;
53    unsigned int n_blocks = ceil((1.0 * SIZE) / threads_per_block);
54    int n = SIZE;
55
56    // Create manager
57    auto manager = kmm::build_runtime();
58    std::deque<kmm::EventId> events;
59
60    for (size_t i = 0; i < 20; i++) {
61        // Request 3 memory areas of a certain size
62        auto A = kmm::Array<float>(n);
63        auto B = kmm::Array<float>(n);
64        auto C = kmm::Array<float>(n);
65
66        // Initialize array A and B on the host
67        manager.submit(kmm::Host(), initialize, write(A), write(B));
68
69        // Execute the function on the device.
70        //manager.submit(kmm::Cuda(), execute, write(C), A, B);
71        manager.submit(kmm::CudaKernel(n_blocks, threads_per_block), vector_add, A, B, write(C), n);
72
73        // Verify the result on the host.
74        auto verify_event = manager.submit(kmm::Host(), verify, C);
75
76        events.push_back(verify_event);
77        if (events.size() >= 5) {
78            manager.wait(events[events.size() - 5]);
79        }
80    }
81
82    manager.synchronize();
83    std::cout << "done\n";
84
85    return 0;
86}
